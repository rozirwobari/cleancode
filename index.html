public class Kubus {
    /**
    * @param panjangSisi panjang sisi kubus.
    * @return Hasil dari volume kubus.
    */
    public static double MenghitungVolume(double panjangSisi) {
        return Math.pow(panjangSisi, 3);
    }
                
    /**
    * @param panjangSisi panjang sisi kubus.
    * @return luas permukaan kubus.
    */
    public static double MenghitungLuasPermukaan(double panjangSisi) {
        return 6 * Math.pow(panjangSisi, 2);
    }
                
    public static void main(String[] args) {
        double panjangSisi = 5.0;
                
        double volume = MenghitungVolume(panjangSisi);
        double luasPermukaan = MenghitungLuasPermukaan(panjangSisi);
                
        System.out.println("Volume kubus dengan sisi " + panjangSisi + " adalah: " + volume);
        System.out.println("Luas permukaan kubus dengan sisi " + panjangSisi + " adalah: " + luasPermukaan);
    }
}
